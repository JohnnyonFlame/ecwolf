actor AngelOfDeath 27
{
	radius 42
	points 5000
	health 1450, 1550, 1650, 2000
	speed 1, 3
	MONSTER
	+AMBUSH
	states
	{
		Spawn:
			ANGL A -1 NOP T_Stand
			stop
		See:
			ANGL A 5 NOP T_Will
			ANGL A 1.5
			ANGL B 4 NOP T_Will
			ANGL C 5 NOP T_Will
			ANGL C 1.5
			ANGL D 4 NOP T_Will
			loop
		Missile:
			ANGL E 5
			ANGL F 10 A_StartAttack
		Launch:
			ANGL E 5 T_Launch
			ANGL E 0 A_Relaunch
			goto Launch
		Tired:
			ANGL G 20 A_Breathing
			ANGL H 20
			ANGL G 20 A_Breathing
			ANGL H 20
			ANGL G 20 A_Breathing
			ANGL H 20
			ANGL G 20 A_Breathing
			goto See
		Death:
			ANGL A .5 A_Fall
			ANGL A .5 A_DeathScream
			ANGL I 5
			ANGL J 5 A_Slurpie
			ANGL KLMNO 5
			ANGL P -1 A_Victory
			stop
	}
}
